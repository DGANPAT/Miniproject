public class MySingleton
{

public static String singleton;
private MySingleton()
{

}

public static MySingleton getSingltonObj()
{
		Syncronised(MySinglton.class){



		if(singleton==null)
		{
			return singleton= new MySingleton();
		}
		else
		{
			return singleton;
		}
}
}

protected Object clone()
	try{
		return singleton;
	}catch
	{
		return null;
	}
protected Object readResolve()
{
	return singleton;
}
