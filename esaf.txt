public class MySingleton
{
private static String singleton;
private MySingleton(){
}
public static MySingleton getSingletonObj(String singleton){
Synchronised(MySingleton.class){

if(singleton==null)
{
	return singleton=new MySingleton();
}
else
{
	return singleton;
}
}
}

protected Object clone()
{
	try{
		return singleton;
	}
	catch(exception)
{
	return null;
}
protected Object readResolve(){
return singleton;
}
