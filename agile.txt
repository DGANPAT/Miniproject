String str = "helloganpatabcdhelloganpat"
int [] freq=new int[str.length];
char[] str1=str.toCharArray();
for(int i=0;i<=str1.length()-1;i++)
{   freq=1;

	for(int j=i+1;j<=str1.length()-1;j++)
	{
		
		if(str1[i]==str1[j])
		{
			freq++;
			str[j]='';
		}
	}
}
for(int i;i<=freq.length;i++)
{
if(str1[i]!='')
{
System.out.println(str[i]+"="+freq[i]);
}
